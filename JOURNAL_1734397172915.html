const stateManager = {
    lastUpdate: Date.now(),
    reconcileState: (clientState, serverState) => {
        return serverState.timestamp > clientState.timestamp 
            ? serverState 
            : clientState;
    },
    updateInterval: 16.67 // 60fps sync
};
